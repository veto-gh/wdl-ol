44 bool IPlugBase::CompareState(const uint8_t* pIncomingState, int startPos)
45 {
46  bool isEqual = true;
47 
48  const double* data = (const double*) pIncomingState + startPos;
49 
50  // dirty hack here because protools treats param values as 32 bit int and in IPlug they are 64bit float
51  // if we memcmp() the incoming state with the current they may have tiny differences due to the quantization
52  for (int i = 0; i < NParams(); i++)
53  {
54  float v = (float) GetParam(i)->Value();
55  float vi = (float) *(data++);
56 
57  isEqual &= (fabsf(v - vi) < 0.00001);
58  }
59 
60  return isEqual;
61 }

179 int IPlugBase::UnserializeParams(const IByteChunk& chunk, int startPos)
180 {
181  TRACE;
182  int i, n = mParams.GetSize(), pos = startPos;
183  ENTER_PARAMS_MUTEX;

187  double v = 0.0;

192  OnParamReset(kPresetRecall);
193  LEAVE_PARAMS_MUTEX;
